Telaffuz skorlama — kendi ML modelin

Kullanıcının telaffuzunu puanlamak için basit bir yaklaşım: kelime benzerliği + fonetik mesafe. 

// Levenshtein distance (string similarity)
int levenshtein(String a, String b) {
  if (a.isEmpty) return b.length;
  if (b.isEmpty) return a.length;
  
  List<List<int>> dp = List.generate(
    a.length + 1, 
    (i) => List.filled(b.length + 1, 0),
  );
  
  for (int i = 0; i <= a.length; i++) dp[i][0] = i;
  for (int j = 0; j <= b.length; j++) dp[0][j] = j;
  
  for (int i = 1; i <= a.length; i++) {
    for (int j = 1; j <= b.length; j++) {
      int cost = a[i-1] == b[j-1] ? 0 : 1;
      dp[i][j] = [
        dp[i-1][j] + 1,
        dp[i][j-1] + 1,
        dp[i-1][j-1] + cost,
      ].reduce((a, b) => a < b ? a : b);
    }
  }
  
  return dp[a.length][b.length];
}

int pronunciationScore(String expected, String actual) {
  final distance = levenshtein(
    expected.toLowerCase(), 
    actual.toLowerCase(),
  );
  final maxLen = [expected.length, actual.length].reduce((a, b) => a > b ? a : b);
  return ((1 - distance / maxLen) * 100).round();
}

SM-2 Spaced Repetition algoritması

Anki'nin de kullandığı SuperMemo-2 algoritması. Unutma eğrisine göre kelimenin ne zaman tekrar edileceğini hesaplar. 

class SM2 {
  // Kullanıcı performansı 0-5 arası (quality)
  // 0: hiç hatırlamıyor, 5: mükemmel
  
  static SM2Result calculate({
    required int quality,
    required double easeFactor,
    required int repetitions,
    required int intervalDays,
  }) {
    if (quality < 3) {
      // Başarısız — en baştan
      return SM2Result(
        easeFactor: easeFactor,
        repetitions: 0,
        intervalDays: 1,
      );
    }
    
    // Yeni ease factor
    double newEf = easeFactor + 
      (0.1 - (5 - quality) * (0.08 + (5 - quality) * 0.02));
    if (newEf < 1.3) newEf = 1.3;
    
    // Yeni interval
    int newInterval;
    int newReps = repetitions + 1;
    
    if (newReps == 1) {
      newInterval = 1;
    } else if (newReps == 2) {
      newInterval = 6;
    } else {
      newInterval = (intervalDays * newEf).round();
    }
    
    return SM2Result(
      easeFactor: newEf,
      repetitions: newReps,
      intervalDays: newInterval,
    );
  }
}

class SM2Result {
  final double easeFactor;
  final int repetitions;
  final int intervalDays;
  
  DateTime get nextReviewDate => 
    DateTime.now().add(Duration(days: intervalDays));
  
  SM2Result({
    required this.easeFactor,
    required this.repetitions,
    required this.intervalDays,
  });
}

// Kullanım
final result = SM2.calculate(
  quality: 4,           // Kullanıcı doğru bildi
  easeFactor: word.easeFactor,
  repetitions: word.repetitions,
  intervalDays: word.intervalDays,
);

word.easeFactor = result.easeFactor;
word.repetitions = result.repetitions;
word.intervalDays = result.intervalDays;
word.nextReview = result.nextReviewDate;

class CefrPredictor {
  // Basit lineer tahmin
  static CefrPrediction predict(User user) {
    const REQUIRED_WORDS = {
      CefrLevel.A1: 500,
      CefrLevel.A2: 1000,
      CefrLevel.B1: 2000,
      CefrLevel.B2: 4000,
      CefrLevel.C1: 8000,
      CefrLevel.C2: 16000,
    };
    
    final currentLevel = _calculateCurrent(user.knownWords);
    final nextLevel = CefrLevel.values[currentLevel.index + 1];
    final wordsNeeded = REQUIRED_WORDS[nextLevel]! - user.knownWords;
    
    // Haftalık kelime öğrenme hızı
    final weeklyRate = user.wordsLearnedLast30Days / 4.3;
    final weeksNeeded = (wordsNeeded / weeklyRate).ceil();
    
    return CefrPrediction(
      currentLevel: currentLevel,
      nextLevel: nextLevel,
      estimatedWeeks: weeksNeeded,
      estimatedDate: DateTime.now().add(Duration(days: weeksNeeded * 7)),
    );
  }
}

Groq rate limit stratejisi

Günlük 14.400 Whisper isteği çok büyük bir sayı (dakikada 10 req). Ama yine de dikkatli olalım:

class GroqRateLimiter {
  static final _queue = Queue<Future Function()>();
  static bool _processing = false;
  static DateTime _lastRequest = DateTime.now();
  static const _minDelay = Duration(milliseconds: 500);
  
  static Future<T> throttled<T>(Future<T> Function() task) async {
    final completer = Completer<T>();
    
    _queue.add(() async {
      final elapsed = DateTime.now().difference(_lastRequest);
      if (elapsed < _minDelay) {
        await Future.delayed(_minDelay - elapsed);
      }
      
      try {
        final result = await task();
        _lastRequest = DateTime.now();
        completer.complete(result);
      } catch (e) {
        completer.completeError(e);
      }
    });
    
    _processQueue();
    return completer.future;
  }
}

class WordEvaluationCache {
  final Isar isar;
  
  Future<Evaluation?> get(String word, String userAudio) async {
    // Aynı kelime + benzer ses için cache kontrol
    final audioHash = _hashAudio(userAudio);
    final cached = await isar.cachedEvaluations
      .filter()
      .wordEqualTo(word)
      .audioHashEqualTo(audioHash)
      .findFirst();
    
    if (cached != null && 
        cached.createdAt.isAfter(DateTime.now().subtract(Duration(days: 7)))) {
      return cached.evaluation;
    }
    return null;
  }
}
